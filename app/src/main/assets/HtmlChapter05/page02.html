1. Maxime È™i minime

A doua numere

cin >> x >> y;
if(x > y) max = x;
else max = y;

cin >> x >> y;
if(x < y) min = x;
else min = y;

A trei numere

cin >> x >> y >> z;
max = x;
if(y > x) max = y;
if(z > x) max = z;

cin >> x >> y >> z;
min = x;
if(y < x) min = y;
if(z < x) min = z;

A n numere(cu numar cunoscut de valori)

cin >> n >> x;
max = x;
for(int i = 2; i <= n; i++)
{
cin >> x;
if(x > max) max = x;
}

cin >> n >> x;
min = x;
for(int i = 2; i <= n; i++)
{
cin >> x;
if(x < min) min = x;
}

A n numere(cu numar necunoscut de valori)

cin >> x;
max = x;
while(x != 0)
{
cin >> x;
if(x > max) max = x;
}

cin >> x;
min = x;
while(x != 0)
{
cin >> x;
if(x < min) min = x;
}
